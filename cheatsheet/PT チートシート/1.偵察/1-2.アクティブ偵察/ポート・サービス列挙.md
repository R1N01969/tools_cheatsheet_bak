## Linux
```sh
# ネットワーク内のホスト存在確認
sudo fping -a -g 192.168.0.0 192.168.255.255
nmap -v -sn 192.168.1.2-255 -oG ping-sweep.txt
grep Up ping-sweep.txt | cut -d " " -f 2

for i in {1..254}; do timeout 1 bash -c "</dev/tcp/172.16.231.$i/445 &>/dev/null" 2>/dev/null && echo "Host 172.16.231.$i is up";done

for i in $(seq 1 254); do nc -zv -w 1 172.16.231.$i 445; done

# ワイルドカード使用可能
sudo nmap -sn 192.168.*.*

sudo nmap -sn 192.168.1.0/24,10.0.0.1-100

# ファイル指定
cat targets.txt
# 192.168.1.0/24
# 192.168.2.0/24
# 10.0.0.1-100
# 172.16.0.0/16
sudo nmap -sn -iL targets.txt

sudo arp-scan -I eth0 192.168.1.0/24
```
